在麒麟高级服务器操作系统 V11（优先选择）上或者 Ubuntu22/24 操作系统上，随着网络流量的激增，如何高效、安全地监控和过滤网络包？传统防火墙或用户空间工具（如 iptables）可能引入延迟和开销，而 eBPF 提供了一种革命性的解决方案：允许用户在内核中运行自定义代码，实现零拷贝、高性能的包处理。这不仅提升了性能，还增强了安全性，特别是在麒麟 OS 或 Ubuntu 强调安全加固的环境中。

#### 实验原理

*   **eBPF 概述**：eBPF（extended Berkeley Packet Filter）起源于 1992 年的 BPF，用于网络包过滤。2014 年扩展为 eBPF，支持更复杂的程序，如循环和函数调用。它在内核中运行字节码，经过验证器确保安全，避免崩溃。
*   **XDP 机制**：XDP 是 eBPF 的一个钩子点，位于网络驱动接收包后、协议栈前。支持三种模式：PASS（继续处理）、DROP（丢弃）、TX（重定向）。在麒麟 OS 或 Ubuntu 上，XDP 可用于高性能网络应用，如 DDoS 防护。
*   **eBPFmap（Maps）**：用于内核与用户空间共享数据，如计数器或配置。
*   **相关麒麟 OS 或 Ubuntu 特性**：麒麟 V11 或 Ubuntu 22/24 集成了 eBPF 支持，通过 libbpf 库和 bcc 工具链。

#### 背景知识

*   **eBPF 历史**：从 BPF 到 eBPF 的发展，关键事件如 2015 年内核合并。
*   **优势**：高性能（接近线速）、安全性（字节码验证）、灵活性（支持多种钩子）。
*   **应用场景**：网络过滤、系统跟踪、容器安全。在麒麟 OS 或 Ubuntu 上，可用于智能运维和安全加固。

#### 核心代码（C 语言和 eBPF）

**`xdp_prog.c` (eBPF 程序)**

```c
#include <linux/bpf.h>
#include <bpf/bpf_helpers.h>
#include <linux/if_ether.h>
#include <linux/ip.h>
#include <linux/tcp.h>
#include <linux/udp.h>

struct packet_stats {
    __u64 total_packets;
    __u64 ip_packets;
    __u64 tcp_packets;
    __u64 udp_packets;
    __u64 dropped_packets;
};

struct {
    __uint(type, BPF_MAP_TYPE_ARRAY);
    __uint(max_entries, 1);
    __type(key, __u32);
    __type(value, struct packet_stats);
} stats_map SEC(".maps");

struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 1024);
    __type(key, __u32);
    __type(value, __u64);
} ip_count SEC(".maps");

struct {
    __uint(type, BPF_MAP_TYPE_PERCPU_ARRAY);
    __uint(max_entries, 1);
    __type(key, __u32);
    __type(value, __u64);
} per_cpu_packets SEC(".maps");

SEC("xdp")
int xdp_prog_simple(struct xdp_md *ctx) {
    void *data_end = (void *)(long)ctx->data_end;
    void *data = (void *)(long)ctx->data;
    struct ethhdr *eth = data;

    __u32 key = 0;
    struct packet_stats *stats = bpf_map_lookup_elem(&stats_map, &key);
    if (stats) {
        __sync_fetch_and_add(&stats->total_packets, 1);
    }

    if ((void*)eth + sizeof(*eth) > data_end)
        return XDP_DROP;

    // 过滤：只允许 IP 包
    if (eth->h_proto == __constant_htons(ETH_P_IP)) {
        struct iphdr *ip = (void*)eth + sizeof(*eth);
        if ((void*)ip + sizeof(*ip) > data_end)
            return XDP_DROP;

        if (stats) {
            __sync_fetch_and_add(&stats->ip_packets, 1);
        }

        // 更新 IP 计数
        __u64 *ip_cnt = bpf_map_lookup_elem(&ip_count, &ip->saddr);
        if (ip_cnt) {
            __sync_fetch_and_add(ip_cnt, 1);
        } else {
            __u64 init_val = 1;
            bpf_map_update_elem(&ip_count, &ip->saddr, &init_val, BPF_ANY);
        }

        // 检查 TCP/UDP
        if (ip->protocol == IPPROTO_TCP) {
            struct tcphdr *tcp = (void*)ip + (ip->ihl * 4);
            if ((void*)tcp + sizeof(*tcp) <= data_end) {
                if (stats) {
                    __sync_fetch_and_add(&stats->tcp_packets, 1);
                }
                // 示例：过滤特定端口
                if (tcp->dest == __constant_htons(80) || tcp->dest == __constant_htons(443)) {
                    return XDP_PASS;
                }
            }
        } else if (ip->protocol == IPPROTO_UDP) {
            struct udphdr *udp = (void*)ip + (ip->ihl * 4);
            if ((void*)udp + sizeof(*udp) <= data_end) {
                if (stats) {
                    __sync_fetch_and_add(&stats->udp_packets, 1);
                }
            }
        }

        // 示例：丢弃特定 IP
        if (ip->saddr == 0xC0A80101) { // 192.168.1.1
            if (stats) {
                __sync_fetch_and_add(&stats->dropped_packets, 1);
            }
            return XDP_DROP;
        }

        return XDP_PASS;
    }

    if (stats) {
        __sync_fetch_and_add(&stats->dropped_packets, 1);
    }
    return XDP_DROP;
}

char _license[] SEC("license") = "GPL";
```

**`loader.c` (用户空间加载器)**

```c
#include <bpf/libbpf.h>
#include <stdio.h>
#include <unistd.h>
#include <net/if.h>
#include <linux/if_link.h>
#include <signal.h>
#include <stdlib.h>
#include <string.h>
#include <arpa/inet.h>

volatile int running = 1;

void sig_handler(int sig) {
    running = 0;
}

int main(int argc, char **argv) {
    struct bpf_object *obj;
    struct bpf_program *prog;
    int prog_fd, ifindex;
    const char *ifname = "eth0";

    if (argc > 1)
        ifname = argv[1];

    ifindex = if_nametoindex(ifname);
    if (!ifindex) {
        perror("if_nametoindex");
        return 1;
    }

    obj = bpf_object__open_file("xdp_prog.o", NULL);
    if (libbpf_get_error(obj)) {
        fprintf(stderr, "Failed to open BPF object\n");
        return 1;
    }

    if (bpf_object__load(obj)) {
        fprintf(stderr, "Failed to load BPF object\n");
        return 1;
    }

    prog = bpf_object__find_program_by_name(obj, "xdp_prog_simple");
    prog_fd = bpf_program__fd(prog);

    if (bpf_xdp_attach(ifindex, prog_fd, XDP_FLAGS_SKB_MODE, NULL)) {
        fprintf(stderr, "Failed to attach XDP program\n");
        return 1;
    }

    printf("XDP program attached to %s. Monitoring traffic...\n", ifname);

    // 获取 mapFD
    struct bpf_map *stats_map = bpf_object__find_map_by_name(obj, "stats_map");
    struct bpf_map *ip_count_map = bpf_object__find_map_by_name(obj, "ip_count");
    int stats_fd = bpf_map__fd(stats_map);
    int ip_fd = bpf_map__fd(ip_count_map);

    signal(SIGINT, sig_handler);

    while (running) {
        sleep(5);

        // 读取统计
        __u32 key = 0;
        struct packet_stats stats;
        if (bpf_map_lookup_elem(stats_fd, &key, &stats) == 0) {
            printf("\n--- Packet Statistics ---\n");
            printf("Total packets: %llu\n", stats.total_packets);
            printf("IP packets: %llu\n", stats.ip_packets);
            printf("TCP packets: %llu\n", stats.tcp_packets);
            printf("UDP packets: %llu\n", stats.udp_packets);
            printf("Dropped packets: %llu\n", stats.dropped_packets);
        }

        // 读取前 10 个 IP 计数
        printf("\n--- Top IP Sources ---\n");
        __u32 ip_key;
        __u64 ip_value;
        void *prev_key = NULL;
        int count = 0;
        while (bpf_map_get_next_key(ip_fd, prev_key, &ip_key) == 0 && count < 10) {
            if (bpf_map_lookup_elem(ip_fd, &ip_key, &ip_value) == 0) {
                struct in_addr addr;
                addr.s_addr = ip_key;
                printf("%s: %llu packets\n", inet_ntoa(addr), ip_value);
            }
            prev_key = &ip_key;
            count++;
        }
    }

    printf("\nDetaching XDP program...\n");
    bpf_xdp_detach(ifindex, XDP_FLAGS_SKB_MODE, NULL);
    bpf_object__close(obj);
    return 0;
}
```

*   **代码说明**：扩展 eBPF 程序为多 map 统计（总计、IP、TCP/UDP 计数、IP 源统计）。用户空间程序实时读取和显示统计数据。
*   **代码说明**：eBPF 程序过滤并计数，用户空间程序加载、监控和分离。编译命令：`clang -O2 -target bpf -c xdp_prog.c -o xdp_prog.o`；运行：`./loader`。
