### 二、简答题

1. 阅读以下程序，写出所有可能的输出结果，并说明理由。
```c
#include <sys/types.h>
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main()
{
    pid_t pid;
    int x = 5;
    pid = fork();
    if (pid == 0) { /* child */
        x += 10;
        printf("child: x = %d\n", x);
    }
    else if (pid > 0) { /* parent */
        x -= 3;
        wait(NULL);
        printf("parent: x = %d\n", x);
    }
    return 0;
}
```
